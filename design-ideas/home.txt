    <script>
        const mockDatabase = [
            { id: 1, title: "Boerenkool Stamppot", time: "45m", taxonomies: { cuisine: ["Dutch"], course: ["Dinner"], occasion: ["Low"] }, image: "https://images.unsplash.com/photo-1604908176997-125f25cc6f3d?auto=format&fit=crop&q=80&w=800" },
            { id: 2, title: "Classic Carbonara", time: "20m", taxonomies: { cuisine: ["Italian"], course: ["Dinner"], occasion: ["Medium"] }, image: "https://images.unsplash.com/photo-1612874742237-6526221588e3?auto=format&fit=crop&q=80&w=800" },
            { id: 3, title: "Sourdough Starter", time: "7d", taxonomies: { cuisine: ["Basics"], course: ["Prep"], occasion: ["High"] }, image: "https://images.unsplash.com/photo-1509440159596-0249088772ff?auto=format&fit=crop&q=80&w=800" },
            { id: 4, title: "Tom Kha Gai", time: "30m", taxonomies: { cuisine: ["Thai"], course: ["Dinner", "Lunch"], occasion: ["Medium"] }, image: "https://images.unsplash.com/photo-1548943487-a2e4e43b485d?auto=format&fit=crop&q=80&w=800" },
            { id: 5, title: "Overnight Oats", time: "5m", taxonomies: { cuisine: ["Global"], course: ["Breakfast"], occasion: ["Low"] }, image: "https://images.unsplash.com/photo-1517673132405-a56a62b18caf?auto=format&fit=crop&q=80&w=800" },
            { id: 6, title: "Beef Wellington", time: "3h", taxonomies: { cuisine: ["British"], course: ["Dinner"], occasion: ["High"] }, image: "https://images.unsplash.com/photo-1600891964092-4316c288032e?auto=format&fit=crop&q=80&w=800" }
        ];

        let currentSearch = "";
        let isFilterOpen = false;
        let activeFilters = { cuisine: [], course: [], occasion: [] };

        const taxonomyOptions = {
            cuisine: [...new Set(mockDatabase.flatMap(r => r.taxonomies.cuisine))].sort(),
            course: [...new Set(mockDatabase.flatMap(r => r.taxonomies.course))].sort(),
            occasion: ["Low", "Medium", "High"]
        };

        const grid = document.getElementById('recipeGrid');
        const filterContainer = document.getElementById('filterContainer');
        const searchInput = document.getElementById('searchInput');
        const resultsCount = document.getElementById('resultsCount');
        const clearFiltersBtn = document.getElementById('clearFiltersBtn');
        const filterToggleBtn = document.getElementById('filterToggleBtn');
        const filterPanel = document.getElementById('filterPanel');
        const filterToggleText = document.getElementById('filterToggleText');

        function init() {
            renderFilters();
            renderGrid();
            setupEventListeners();
        }

        function setupEventListeners() {
            searchInput.addEventListener('input', (e) => {
                currentSearch = e.target.value.toLowerCase();
                renderGrid();
            });

            clearFiltersBtn.addEventListener('click', () => {
                activeFilters = { cuisine: [], course: [], occasion: [] };
                renderFilters();
                renderGrid();
            });

            filterToggleBtn.addEventListener('click', () => {
                isFilterOpen = !isFilterOpen;
                if (isFilterOpen) {
                    filterPanel.classList.remove('hidden');
                    filterToggleText.textContent = 'Close';
                    filterToggleBtn.classList.replace('bg-black', 'bg-gray-200');
                    filterToggleBtn.classList.replace('text-white', 'text-black');
                } else {
                    filterPanel.classList.add('hidden');
                    filterToggleText.textContent = 'Filters';
                    filterToggleBtn.classList.replace('bg-gray-200', 'bg-black');
                    filterToggleBtn.classList.replace('text-black', 'text-white');
                }
            });
        }

        function handleFilterClick(category, value) {
            const index = activeFilters[category].indexOf(value);
            if (index > -1) {
                activeFilters[category].splice(index, 1);
            } else {
                activeFilters[category].push(value);
            }
            renderFilters();
            renderGrid();
        }

        function renderFilters() {
            let html = '';
            for (const [category, options] of Object.entries(taxonomyOptions)) {
                html += `
                    <div>
                        <h3 class="text-[10px] font-bold text-gray-400 uppercase tracking-widest mb-3">${category}</h3>
                        <div class="flex flex-wrap gap-2">
                            ${options.map(option => {
                                const isActive = activeFilters[category].includes(option);
                                const baseClasses = "px-3 py-1 text-xs font-medium cursor-pointer transition-colors border";
                                const activeClasses = isActive 
                                    ? "bg-black text-white border-black" 
                                    : "bg-white text-black border-gray-300 hover:border-black";
                                return `<button class="${baseClasses} ${activeClasses}" onclick="handleFilterClick('${category}', '${option}')">${option}</button>`;
                            }).join('')}
                        </div>
                    </div>
                `;
            }
            filterContainer.innerHTML = html;
        }

        function renderGrid() {
            const filteredRecipes = mockDatabase.filter(recipe => {
                const matchesSearch = recipe.title.toLowerCase().includes(currentSearch);
                const matchesTaxonomies = Object.keys(activeFilters).every(category => {
                    const selected = activeFilters[category];
                    if (selected.length === 0) return true;
                    const tags = recipe.taxonomies[category] || [];
                    return selected.some(s => tags.includes(s));
                });
                return matchesSearch && matchesTaxonomies;
            });

            resultsCount.textContent = `${filteredRecipes.length} entries`;
            
            const hasActiveFilters = Object.values(activeFilters).some(arr => arr.length > 0);
            clearFiltersBtn.style.display = hasActiveFilters ? 'block' : 'none';

            if (filteredRecipes.length === 0) {
                grid.innerHTML = `
                    <div class="col-span-full py-20 text-center">
                        <p class="text-sm font-bold uppercase tracking-widest text-gray-400">Null results.</p>
                    </div>`;
                return;
            }

            grid.innerHTML = filteredRecipes.map(recipe => {
                const displayTags = [...recipe.taxonomies.cuisine, ...recipe.taxonomies.course, ...recipe.taxonomies.effort];
                
                return `
                <article class="recipe-card group cursor-pointer">
                    <div class="image-container mb-4">
                        <img src="${recipe.image}" alt="${recipe.title}" loading="lazy">
                    </div>
                    <div>
                        <div class="flex justify-between items-baseline mb-1">
                            <h2 class="text-lg font-black tracking-tight">${recipe.title}</h2>
                            <span class="text-xs font-bold text-gray-400">${recipe.time}</span>
                        </div>
                        <div class="flex flex-wrap gap-x-2 gap-y-1 mt-2">
                            ${displayTags.map(tag => `<span class="text-[10px] text-gray-500 font-bold uppercase tracking-wider">${tag}</span>`).join('')}
                        </div>
                    </div>
                </article>
            `}).join('');
        }

        init();
    </script>
